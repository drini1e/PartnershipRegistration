    setHidden(wrapDesc,false);

  }

  }

function syncEstablishmentDate(){
  var date = document.getElementById("establishmentDate").value;
  if(!date) return;

  var parts = date.split("-");
  var year = parts[0];
  var month = parseInt(parts[1], 10);
  var day = parseInt(parts[2], 10);

  FieldWithName(document.ProfileForm,"inp_11563[]",0).innerHTML = '<option value="' + day + '" selected>' + day + '</option>';
  FieldWithName(document.ProfileForm,"inp_11563[]",1).innerHTML = '<option value="' + month + '" selected>' + month + '</option>';
  FieldWithName(document.ProfileForm,"inp_11563[]",2).value = year;
}
